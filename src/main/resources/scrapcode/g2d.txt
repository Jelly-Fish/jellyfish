    public void addEventEffect(final Color c, final int borderWidth) {
        
        final Color color = new Color(c.getRed(), c.getGreen(), c.getBlue(), 150);
                
        
        double x = 0.0;
        double y = 0.0;
        try {
            x = this.getLocationOnScreen().getX();
            y = this.getLocationOnScreen().getY();
        } catch (Exception ex) {
            System.out.println("catched :S");
            return;
        }
        System.out.println("x=" + x + " y=" + y);
        Graphics2D g2d = (Graphics2D) this.getGraphics();
        Shape shape = new Rectangle2D.Double(x, y, this.getWidth(), this.getHeight());
        g2d.setColor(color);
        g2d.draw(shape);
        g2d.fill(new Rectangle2D.Double(this.getX(), this.getY(), this.getWidth(), this.getHeight()));
        
        this.setBorder(BorderFactory.createLineBorder(c, borderWidth));
        this.effectDisplayed = true;
        this.repaint();
    }
